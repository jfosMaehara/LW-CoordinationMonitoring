
SELECT
	受信取込管理番号
,	枝番
,	定義名
,	事業所CD
,	荷主CD
,	荷主部門CD
,	受注先CD
,	連携TBL名
,	連携MDB_FULL_PATH = 連携FILEPATH + 連携FILE名 + 連携FILE拡張子

FROM
	LJCDB0101.dbo.LM受信取込定義 AS P
WHERE
	EXISTS 
	(
		SELECT
			*
		FROM
			EPCDB0101.dbo.EJ受注出荷売上 AS C
		WHERE
			事業所CD = '01'
		AND	PIC中状況FLG = '1'
		AND 出荷待状況FLG= '1'
		AND 売上区分 <> '9'
		AND 出荷確定状況FLG = '0'
		AND (
					直送区分 NOT IN ('1', '3') 
				OR	(
							直送区分='2' 
						AND	振替出荷事業所CD IS NULL
					)
			)
		AND NOT EXISTS 
			(
				SELECT
					*
				FROM
					LJCDB0101.dbo.LJ受注連携累積V2 AS G
				WHERE
					G.管理番号 = C.管理番号
				AND G.管理行番号 = C.管理行番号
			)
		AND P.大荷主CD = '001'
		AND P.荷主CD = '001'
		AND P.荷主部門CD = '01'
		AND P.事業所CD = C.事業所CD
		AND P.受注先CD = C.お客様CD
		AND RIGHT(P.連携TBL名, 4 ) = C.請求先CD --請求先とテーブル名が違って登録されている可能性がある。 その場合この条件は排除する必要があるが今のところはないはず。
	)
